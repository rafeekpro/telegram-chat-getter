### Dependencies (pyproject.toml)
```toml

name = "telegram-getter"
version = "0.1.0"
requires-python = ">=3.11"
dependencies = [
    "telethon>=1.34.0",
    "aiofiles>=23.2.1",
    "python-dotenv>=1.0.0",
    "rich>=13.7.0",
]

[project.optional-dependencies]
dev = [
    "pytest>=7.4.0",
    "pytest-asyncio>=0.23.0",
]

[project.scripts]
telegram-getter = "telegram_getter.cli:main"
```
